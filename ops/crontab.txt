# ============================================================================
# NurEine Mac-Mini-Runner — crontab (ALLE ZEITEN UTC)
# ----------------------------------------------------------------------------
# Installieren:  bash ops/install-cron.sh
# Ansehen:       crontab -l
# Voraussetzung: cron läuft in UTC. Prüfen: `timedatectl` (RTC/zeitzone).
#   Wir halten die Job-Zeiten in UTC (wie die bisherigen GitHub-Workflows).
#   Falls die Mini-Systemzeit auf Europe/Berlin steht, sind cron-Zeiten LOKAL —
#   dann install-cron.sh nutzt `CRON_TZ=UTC` (unten gesetzt), damit es UTC bleibt.
#
# Prinzip: gestaffelt, KEIN Overlap schwerer Jobs (der Mini fährt nie zwei
# schwere Sachen gleichzeitig). Kette fixt die Timing-Bugs aus
# docs/NACHT_ROUTINEN_PROBLEME.md (strikt, alles nach 03:00, mit Abstand).
# ============================================================================
CRON_TZ=UTC
SHELL=/bin/bash

# --- NACHT-KETTE (Claude-Agenten, gestaffelt, kein Overlap) -----------------
# Analyst zuerst (leicht) — wertet Vortag aus, bevor neu gefetcht wird.
30 1 * * *   $HOME/NurEine/ops/run/agent.sh analyst
# Fetch: RSS + Vorfilter + Claude-Analyse (--export/--import). Schwer.
30 2 * * *   $HOME/NurEine/ops/run/agent.sh fetch
# Chefredakteur: bewertet neu, wählt Tages-Perlen. NACH Fetch.
15 3 * * *   $HOME/NurEine/ops/run/agent.sh chefredakteur
# Veredler: schreibt die Perlen-Felder aus. NACH Chefredakteur.
50 3 * * *   $HOME/NurEine/ops/run/agent.sh veredler
# Bild-Regie: fal.ai für die Perlen. NACH Veredler. Schwer (Bildgen + QA).
20 4 * * *   $HOME/NurEine/ops/run/agent.sh bildregie
# Verbesserer: Code-Verbesserungen (PRs/Commits). Allein, kein Overlap.
0 5 * * *    $HOME/NurEine/ops/run/agent.sh verbesserer
# Reel-/TikTok-Regie: Remotion-Render (schwer, Chromium). Mo/Di/Mi/Fr/Sa, allein.
30 6 * * 1,2,3,5,6   $HOME/NurEine/ops/run/reel.sh

# --- THIN-TRIGGER (curl auf /api/cron/*, leicht) ----------------------------
30 5 * * *   $HOME/NurEine/ops/run/trigger.sh social-publish
15 6 * * *   $HOME/NurEine/ops/run/trigger.sh social-generate
30 6 * * *   $HOME/NurEine/ops/run/trigger.sh highlight
15 7 * * *   $HOME/NurEine/ops/run/trigger.sh indexnow '{"recent":50}'
15 8 * * *   $HOME/NurEine/ops/run/trigger.sh social-threads
0 22 * * *   $HOME/NurEine/ops/run/trigger.sh social-insights
# social-comments 4×/Tag
30 7,11,15,19 * * *   $HOME/NurEine/ops/run/trigger.sh social-comments
# social-story stündlich 07–21 UTC
0 7-21 * * *   $HOME/NurEine/ops/run/trigger.sh social-story
# social-digest sonntags
30 15 * * 0   $HOME/NurEine/ops/run/trigger.sh social-digest

# --- COMPUTE-JOBS (Python, selten) ------------------------------------------
# World-Metrics am 1. des Monats
0 5 1 * *    $HOME/NurEine/ops/run/pyjob.sh scripts/fetch_world_metrics.py
# Worldbank montags (läuft aus scripts/)
30 5 * * 1   $HOME/NurEine/ops/run/pyjob.sh fetch_worldbank.py --cwd scripts
# World-Newsletter am 1. des Monats (Thin-Trigger nach Metrik-Fetch)
20 6 1 * *   $HOME/NurEine/ops/run/trigger.sh world-newsletter
